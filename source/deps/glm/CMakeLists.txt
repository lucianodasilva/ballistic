set(NAME glm)

file(GLOB ROOT_SOURCE ${CMAKE_CURRENT_LIST_DIR}/*.cpp)
file(GLOB ROOT_INLINE ${CMAKE_CURRENT_LIST_DIR}/*.inl)
file(GLOB ROOT_HEADER ${CMAKE_CURRENT_LIST_DIR}/*.hpp)

file(GLOB_RECURSE CORE_SOURCE ${CMAKE_CURRENT_LIST_DIR}/core/*.cpp)
file(GLOB_RECURSE CORE_INLINE ${CMAKE_CURRENT_LIST_DIR}/core/*.inl)
file(GLOB_RECURSE CORE_HEADER ${CMAKE_CURRENT_LIST_DIR}/core/*.hpp)

file(GLOB_RECURSE GTC_SOURCE ${CMAKE_CURRENT_LIST_DIR}/gtc/*.cpp)
file(GLOB_RECURSE GTC_INLINE ${CMAKE_CURRENT_LIST_DIR}/gtc/*.inl)
file(GLOB_RECURSE GTC_HEADER ${CMAKE_CURRENT_LIST_DIR}/gtc/*.hpp)

file(GLOB_RECURSE GTX_SOURCE ${CMAKE_CURRENT_LIST_DIR}/gtx/*.cpp)
file(GLOB_RECURSE GTX_INLINE ${CMAKE_CURRENT_LIST_DIR}/gtx/*.inl)
file(GLOB_RECURSE GTX_HEADER ${CMAKE_CURRENT_LIST_DIR}/gtx/*.hpp)

file(GLOB_RECURSE VIRTREV_SOURCE ${CMAKE_CURRENT_LIST_DIR}/virtrev/*.cpp)
file(GLOB_RECURSE VIRTREV_INLINE ${CMAKE_CURRENT_LIST_DIR}/virtrev/*.inl)
file(GLOB_RECURSE VIRTREV_HEADER ${CMAKE_CURRENT_LIST_DIR}/virtrev/*.hpp)

source_group("Core Files" FILES ${CORE_SOURCE})
source_group("Core Files" FILES ${CORE_INLINE})
source_group("Core Files" FILES ${CORE_HEADER})
source_group("GTC Files" FILES ${GTC_SOURCE})
source_group("GTC Files" FILES ${GTC_INLINE})
source_group("GTC Files" FILES ${GTC_HEADER})
source_group("GTX Files" FILES ${GTX_SOURCE})
source_group("GTX Files" FILES ${GTX_INLINE})
source_group("GTX Files" FILES ${GTX_HEADER})
source_group("VIRTREV Files" FILES ${VIRTREV_SOURCE})
source_group("VIRTREV Files" FILES ${VIRTREV_INLINE})
source_group("VIRTREV Files" FILES ${VIRTREV_HEADER})

include_directories(${CMAKE_CURRENT_LIST_DIR})

add_executable( glm 
	${ROOT_SOURCE}    ${ROOT_INLINE}    ${ROOT_HEADER} 
	${CORE_SOURCE}    ${CORE_INLINE}    ${CORE_HEADER}
	${GTC_SOURCE}     ${GTC_INLINE}     ${GTC_HEADER} 
	${GTX_SOURCE}     ${GTX_INLINE}     ${GTX_HEADER} 
	${VIRTREV_SOURCE} ${VIRTREV_INLINE} ${VIRTREV_HEADER})
